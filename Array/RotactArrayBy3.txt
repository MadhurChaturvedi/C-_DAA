#include <bits/stdc++.h> 
using namespace std;

int main() 
{
    vector<int>arr = {1,2,3,4,5};
    int d=3;
    int n=arr.size();

    // cout<<n;
    // Storing into the Temp vector
    vector<int> Temp(d);
    
    for(int i=0;i<d;i++){
        Temp[i] = arr[i];
        // cout<<Temp[i];
    }

    // shifting to the left remaning element by one's
    for(int i=d;i<n;i++){
        arr[i-d] = arr[i];
    }  

    // marge the temp array
    for(int i=n-d;i<n;i++){
        arr[i] = Temp[i-(n-d)];
    }
    
    for(int i:arr){
        cout<<i<<" ";
    }
    return 0;
}
