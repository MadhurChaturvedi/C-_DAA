#include<bits/stdc++.h>
using namespace std;
int main() 
{

    vector<int> arr={1,0,2,3,2,0,0,4,5};
    vector<int> temp;
    for(int i=0;i<arr.size();i++){
       if(arr[i] != 0){
        temp.push_back(arr[i]);
       }
    }
    for(int i=0;i<arr.size();i++){
        arr[i] = temp[i];
    }
    int ts = temp.size();
    for(int i=ts;i<arr.size();i++){
       arr[i] = 0;
    }    

    for(int i : arr){
        cout<< i <<" ";
    }
    return 0;
}
