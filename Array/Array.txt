// secound Largest and Smallest 
#include<iostream>
#include<vector>
using namespace std;

int secoundLargest(vector<int>&a){
 int largest = a[0];
 int secound = -1;
 for(int i=0;i<a.size();i++){
     if(a[i]>largest){
          secound = largest;
          largest = a[i];
       }
     else if(a[i]<largest && a[i]>secound){
       secound = a[i];
  } 	
   }
return secound;
}
