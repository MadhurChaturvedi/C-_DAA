#include<iostream>
using namespace std;

void f(int i,int n){
  if(i<n) return;
  f(i-1,n);
  cout<<i<<" ";
}

int main(){
int i=5;
int n=1;
cout<<"Tesing Main"<<endl;
f(i,n);
return 0;
}

#include<iostream>
using namespace std;

int f(int n){
  if(n<=1){
     return 1;
  } 

  return n*f(n-1);
}

int main(){
int n=3;
cout<<"Tesing Main"<<endl;
cout<< f(n);
return 0;
}
