#include<iostream>
#include<utlitys>
using namespace std;


void Selection_Sort(int arr[],int n){
   int mini=i;
   for(int i=0;i<=n-2;i++){
       for(int j=i;j<=n-1;j++){
           if(arr[j] < arr[mini]){
              mini = j;
           }
        } 
    swap(arr[i],arr[mini]);
   }

}
